package xsbti.api;

import java.util.Arrays;
import java.util.List;
public class Modifiers implements java.io.Serializable
{
	public Modifiers(boolean isAbstract, boolean isDeferred, boolean isOverride, boolean isFinal, boolean isSealed, boolean isImplicit, boolean isLazy, boolean isSynthetic)
	{
		this.isAbstract = isAbstract;
		this.isDeferred = isDeferred;
		this.isOverride = isOverride;
		this.isFinal = isFinal;
		this.isSealed = isSealed;
		this.isImplicit = isImplicit;
		this.isLazy = isLazy;
		this.isSynthetic = isSynthetic;
	}
	private final boolean isAbstract;
	private final boolean isDeferred;
	private final boolean isOverride;
	private final boolean isFinal;
	private final boolean isSealed;
	private final boolean isImplicit;
	private final boolean isLazy;
	private final boolean isSynthetic;
	public final boolean isAbstract()
	{
		return isAbstract;
	}
	public final boolean isDeferred()
	{
		return isDeferred;
	}
	public final boolean isOverride()
	{
		return isOverride;
	}
	public final boolean isFinal()
	{
		return isFinal;
	}
	public final boolean isSealed()
	{
		return isSealed;
	}
	public final boolean isImplicit()
	{
		return isImplicit;
	}
	public final boolean isLazy()
	{
		return isLazy;
	}
	public final boolean isSynthetic()
	{
		return isSynthetic;
	}
	public String toString()
	{
		return "Modifiers(" + "isAbstract: " + isAbstract() + ", " + "isDeferred: " + isDeferred() + ", " + "isOverride: " + isOverride() + ", " + "isFinal: " + isFinal() + ", " + "isSealed: " + isSealed() + ", " + "isImplicit: " + isImplicit() + ", " + "isLazy: " + isLazy() + ", " + "isSynthetic: " + isSynthetic()+ ")";
	}
}
