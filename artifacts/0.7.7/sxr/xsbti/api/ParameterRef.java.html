package xsbti.api;

import java.util.Arrays;
import java.util.List;
public class ParameterRef extends SimpleType 
{
	public ParameterRef(int id)
	{
		this.id = id;
	}
	private final int id;
	public final int id()
	{
		return id;
	}
	public String toString()
	{
		return "ParameterRef(" + "id: " + id()+ ")";
	}
}
